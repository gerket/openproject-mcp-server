# OpenProject MCP Server — Environment Variables
# Copy this file to .env and fill in your values.

# ── Required ──────────────────────────────────────────────────────────────────

# Base URL of your OpenProject instance
OPENPROJECT_URL=https://your-instance.openproject.com

# API token — User menu → Account settings → Access tokens → + API Token
OPENPROJECT_API_KEY=your-api-token-here

# ── Tool filtering ────────────────────────────────────────────────────────────

# Expose only tools matching these tags (comma-separated, OR/union).
# Tools are tagged by endpoint resource (work-packages, projects, versions,
# time-entries, …) plus a <resource>-read / <resource>-write composite.
# Examples: work-packages | work-packages,projects | versions-read,time-entries-write
# OPENPROJECT_MCP_INCLUDE_TAGS=work-packages,projects

# Hide tools matching these tags (comma-separated). Applied after include.
# Useful examples:
#   write              — read-only server (hide every mutating tool)
#   admin              — hide admin-only tools when using a non-admin token
#   storages,file-links — hide file tools on instances without OneDrive/Nextcloud
# OPENPROJECT_MCP_EXCLUDE_TAGS=write

# ── Optional ──────────────────────────────────────────────────────────────────

# HTTP proxy URL
# OPENPROJECT_PROXY=http://proxy.company.com:8080

# Path to custom CA bundle (PEM) for self-signed TLS certificates
# OPENPROJECT_CA_BUNDLE=/path/to/ca-bundle.pem

# Log level: DEBUG | INFO | WARNING | ERROR (default: INFO)
# LOG_LEVEL=INFO

# ── Integration tests only ────────────────────────────────────────────────────

# Project slug for integration tests (default: mcp-test)
# OPENPROJECT_PROJECT=mcp-test

# Bot user API token — required for test_mark_single_notification_read
# Generate by logging in as mcp-test-bot → Account settings → Access tokens
# OPENPROJECT_BOT_API_KEY=mcp-test-bot-token-here

# Seed work package ID — set automatically by scripts/setup_test_project.py
# OPENPROJECT_SEED_WP_ID=142

# Custom action ID for test_execute_custom_action (default: 1)
# OPENPROJECT_CUSTOM_ACTION_ID=1
